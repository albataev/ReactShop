
Категории:

const CategoriesSchema = new Schema({
  title: {
    type: String,
    required: true
  },
  russtitle: {
    type: String,
    required: true
  },
  url: {
    type: String
  },
  description: {
    type: String
  },
  image: {
    type: String
  },
  parent: {
    type: String
  },
  level: {
    type: String
  }
});

Parent - если указано, то размещать ее вторым уровнем в меню
Level - если указано, то видеть 0, 1, 2 и тд, понимать на каком уровне отображать

const CatalogSchema = new Schema({
  title: {
    type: String,
    required: true
  },

    type: String,
    required: true
  },
  price: {
    type: Number,
    required: true
  },

    default: false
  },

    type: String
  },
  longtitle: {
    type: String
  },

    type: String
  },

    type: String
  }
});
